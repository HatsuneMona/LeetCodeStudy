/**
 * Definition for singly-linked list.
 * type ListNode struct {
 *     Val int
 *     Next *ListNode
 * }
 */
func middleNode(head *ListNode) *ListNode {
	// head == nil 不加，因为 “给定链表的结点数介于 1 和 100 之间”
	if head.Next == nil {
		return head
	}

	mid, end := head, head.Next

	for {
		mid, end = mid.Next, end.Next
		if end == nil || end.Next == nil {
			return mid
		}
		end = end.Next
	}

	panic("err")
}
